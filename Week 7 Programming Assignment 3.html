import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# === Parameters ===
stock_symbol = "AAPL"
start_date = "2020-01-01"
end_date = "2023-12-31"
initial_cash = 10000
risk_tolerance = "Moderate"  # Choose from: Low, Moderate, High

# === Step 1: Download data (use adjusted close) ===
df_raw = yf.download(stock_symbol, start=start_date, end=end_date, auto_adjust=True)
df = pd.DataFrame(df_raw['Close']).copy()
df.columns = ['Close']
df.dropna(inplace=True)

# === Step 2: Feature Engineering ===
df['SMA20'] = df['Close'].rolling(window=20).mean()
df['score'] = 100 * ((df['Close'] - df['SMA20']) / df['SMA20'])

# === Step 2.5: Compute RSI ===
def compute_rsi(series, period=14):
    delta = series.diff()
    gain = delta.clip(lower=0)
    loss = -delta.clip(upper=0)

    avg_gain = gain.rolling(window=period).mean()
    avg_loss = loss.rolling(window=period).mean()

    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

df['RSI'] = compute_rsi(df['Close'], period=14)
df.dropna(inplace=True)

# === Step 3: Adjusted thresholds ===
thresholds = {
    "Low": {"buy": 2.0, "sell": -2.0},
    "Moderate": {"buy": 1.5, "sell": -1.5},
    "High": {"buy": 1.0, "sell": -1.0}
}

# Enhanced decision rule using RSI
def decide_action(score, rsi, risk_level):
    t = thresholds[risk_level]
    if score >= t["buy"] and rsi < 70:
        return "Buy"
    elif score <= t["sell"] and rsi > 30:
        return "Sell"
    else:
        return "Hold"

df['action'] = df.apply(lambda row: decide_action(row['score'], row['RSI'], risk_tolerance), axis=1)

# === Step 4: Simulate trading ===
cash = initial_cash
position = 0
portfolio_values = []

for date, row in df.iterrows():
    action = row['action']
    price = row['Close']

    if action == "Buy" and cash >= price:
        position = cash // price
        cash -= position * price
        print(f"[{date.date()}] BUY {position} shares @ ${price:.2f}")

    elif action == "Sell" and position > 0:
        cash += position * price
        print(f"[{date.date()}] SELL {position} shares @ ${price:.2f}")
        position = 0

    portfolio_value = cash + position * price
    portfolio_values.append(portfolio_value)

df['portfolio_value'] = portfolio_values

# === Step 5: Plot performance ===
plt.figure(figsize=(12, 6))
plt.plot(df.index, df['portfolio_value'], label='Portfolio Value')
plt.plot(df.index, df['Close'] / df['Close'].iloc[0] * initial_cash, label='Buy & Hold')
plt.title(f"Trading Strategy vs Buy & Hold ({stock_symbol})")
plt.ylabel("Portfolio Value ($)")
plt.legend()
plt.grid()
plt.show()

# === Step 6: Final Performance Metrics ===
final_value = df['portfolio_value'].iloc[-1]
buy_hold_value = df['Close'].iloc[-1] / df['Close'].iloc[0] * initial_cash

print("\n--- Performance Summary ---")
print(f"Final Portfolio Value: ${final_value:.2f}")
print(f"Buy & Hold Value:     ${buy_hold_value:.2f}")
print(f"Outperformance:       ${final_value - buy_hold_value:.2f}")
